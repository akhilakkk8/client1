import { useState, useEffect, useRef } from 'react'; export function useDataFetcher<T>( url: string, deps: unknown[] ) { const [data, setData] = useState<T | null>(null); const [loading, setLoading] = useState(true); const [error, setError] = useState<Error | null>(null); const abortRef = useRef<AbortController>(); useEffect(() => { abortRef.current?.abort(); abortRef.current = new AbortController(); fetchData(url, abortRef.current); return () => abortRef.current?.abort(); }, [url, ...deps]); return { data, loading, error }; }

useDataFetcher.ts
Summary
TL;DR A generic React hook that fetches data from a URL, tracks loading and error states, and automatically cancels in-flight requests on re-render or unmount.
Key Points
1
Generic hook — accepts a typed URL and dependency array, returning { data, loading, error }.
2
Abort control — uses AbortController via a ref to cancel stale requests before each new fetch.
3
Cleanup on unmount — the useEffect teardown aborts any pending request, preventing memory leaks.
